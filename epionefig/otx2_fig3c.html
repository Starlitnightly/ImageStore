3 · Build TSS tables from GENCODE¶

epi.utils.read_gtf parses the GTF; we then restrict to protein-coding genes and pick the canonical TSS (start or end depending on strand).

In [3]:
gtf = epi.utils.get_gene_annotation(str(GENCODE_GTF))  # protein_coding, no chrM
gtf['tss'] = np.where(gtf['strand'] == '+', gtf['start'], gtf['end'])
gtf = gtf[~gtf['chrom'].str.contains('_')]

def _tss_table(gene_set):
    sub = gtf[gtf['gene_name'].isin(gene_set)].drop_duplicates('gene_name')
    return sub.rename(columns={'chrom':'seqname'})[
        ['seqname','start','end','strand','gene_name']].assign(
        gene_id=sub['gene_name'].values, feature='transcript')

tgt_df = _tss_table(activated_4c).reset_index(drop=True)
non_df = _tss_table(nonmaternal).reset_index(drop=True)
print(f'resolved TSSs: targets={len(tgt_df)}  nontargets={len(non_df)}')
