OTX2 case study — Fig 3f¶

Violin of log2(KD/control) RNA fold-change on day 2, for 4C-activated genes vs an expression-matched Random control set, each split by whether the gene's promoter (TSS ± 2.5 kb) carries a 4C OTX2 peak.

Paper (two-sided Mann-Whitney U): 4C+ vs 4C- P = 2 × 10⁻⁵; Random+ vs Random- P = 0.8.

Paper: Wang et al. Nat. Genet. 57, 2772-2784 (2025) 10.1038/s41588-025-02350-8

%%script false --no-raise-error
# ==========================================================
# CUT&RUN FASTQ -> deduplicated BAM -> RPKM-normalised bigwig
# ==========================================================
# Reference: Wang et al. 2025 Methods ("CUT&RUN assay" section) +
# filename convention in OMIX006476 (`..._pmu_100bp_rpkm_bamCoverage.bw`).
#
# Raw FASTQs are hosted behind controlled access at GSA-Human HRA006621
# (paper Data Availability). For each replicate:
SAMPLE=h_OTX2_day2_CUTRUN_rep1         # one of rep1 / rep2 / KD etc.
FQDIR=/scratch/users/steorra/data/otx2/HRA006621/fastq
BAMDIR=/scratch/users/steorra/data/otx2/OMIX006476/bam
BWDIR=/scratch/users/steorra/data/otx2/OMIX006476
IDX=/scratch/users/steorra/data/hg19/bt2_idx/hg19
mkdir -p $BAMDIR $BWDIR

# 1. Adapter / quality trim
trim_galore --paired --gzip -o $BAMDIR/trim \
    $FQDIR/${SAMPLE}_R1.fastq.gz $FQDIR/${SAMPLE}_R2.fastq.gz

# 2. Bowtie2 alignment (paired, CUT&RUN-optimised)
bowtie2 --end-to-end --very-sensitive --no-mixed --no-discordant \
        -I 10 -X 700 -p 8 -x $IDX \
        -1 $BAMDIR/trim/${SAMPLE}_R1_val_1.fq.gz \
        -2 $BAMDIR/trim/${SAMPLE}_R2_val_2.fq.gz |
    samtools view -bS -q 30 -f 2 - |
    samtools sort -@ 8 -o $BAMDIR/${SAMPLE}.raw.bam

# 3. Picard MarkDuplicates -> "pmu" (picard mark unique) BAM
picard MarkDuplicates \
    I=$BAMDIR/${SAMPLE}.raw.bam \
    O=$BAMDIR/${SAMPLE}.pmu.bam \
    M=$BAMDIR/${SAMPLE}.pmu.metrics REMOVE_DUPLICATES=true
samtools index $BAMDIR/${SAMPLE}.pmu.bam

# 4. Merge biological replicates (rep1 + rep2 -> rep12)
samtools merge -@ 8 -o $BAMDIR/h_OTX2_day2_CUTRUN_rep12.pmu.bam \
    $BAMDIR/h_OTX2_day2_CUTRUN_rep1.pmu.bam \
    $BAMDIR/h_OTX2_day2_CUTRUN_rep2.pmu.bam
samtools index $BAMDIR/h_OTX2_day2_CUTRUN_rep12.pmu.bam

# 5. RPKM bigwig (matches published file name)
bamCoverage -b $BAMDIR/h_OTX2_day2_CUTRUN_rep12.pmu.bam \
    -o $BWDIR/h_OTX2_day2_CUTRUN_rep12_pmu_100bp_rpkm_bamCoverage.bw \
    --binSize 100 --normalizeUsing RPKM -p 8

# ==========================================================
# RNA-seq FASTQ -> gene counts -> DESeq2 DEG (Supp Table S6)
# ==========================================================
# 1. STAR alignment (hg19 + GENCODE v41lift37)
STAR --runThreadN 8 --runMode alignReads \
     --genomeDir /scratch/users/steorra/data/hg19/star_idx \
     --readFilesIn $FQDIR/${SAMPLE}_R1.fastq.gz $FQDIR/${SAMPLE}_R2.fastq.gz \
     --readFilesCommand zcat --outSAMtype BAM SortedByCoordinate \
     --outFileNamePrefix $BAMDIR/${SAMPLE}.

# 2. featureCounts -> per-sample gene count matrix
featureCounts -T 8 -p --countReadPairs \
     -a /scratch/users/steorra/data/hg19/gencode.v41lift37.annotation.gtf.gz \
     -o counts.tsv $BAMDIR/*.Aligned.sortedByCoord.out.bam

# 3. DESeq2 (R) — below is the R one-liner the paper uses; produces Supp Table S6
#    via contrasts OTX2_KD_vs_ctrl on each stage (D2, D3).
Rscript -e '
library(DESeq2)
counts <- as.matrix(read.table("counts.tsv", header=TRUE, row.names=1,
                                check.names=FALSE, skip=1)[,-(1:5)])
coldata <- data.frame(
    row.names = colnames(counts),
    condition = factor(sub("_rep.*$", "", colnames(counts)))
)
dds <- DESeqDataSetFromMatrix(counts, coldata, ~ condition)
dds <- DESeq(dds)
res <- results(dds, contrast=c("condition","h_OTX2_KD_day2","h_OTX2_ctrl_day2"))
write.table(as.data.frame(res), "S6_D2_OTX2_KD_DEG.tsv",
            sep="\t", quote=FALSE)
'

s4 = pd.read_excel(SUPP_TABLES, sheet_name='Table S4_OTX2 KD_RNA count', header=2)
s4.columns = [str(c).strip() for c in s4.columns]
def _b1(c, pfx):
    m = re.search(r'_rep[12]_(\d[A-Z]\d+)_', c)
    return bool(m) and m.group(1).startswith(pfx)
ctrl = [c for c in s4.columns if 'ctrl_day2' in c and _b1(c, '4C')]
kd   = [c for c in s4.columns if 'KD_day2'   in c and _b1(c, '4K') and 'rescue' not in c]
ct = s4[['gene']+ctrl+kd].copy()
for c in ctrl+kd: ct[c] = pd.to_numeric(ct[c], errors='coerce').fillna(0)
mat = ct[ctrl+kd].values.astype(np.float64); logm = np.log(mat + 1e-9)
geo = np.where(np.all(mat>0,axis=1), np.mean(np.where(mat>0,logm,np.nan),axis=1), np.nan)
mask = ~np.isnan(geo)
sf = np.exp(np.median(logm[mask] - geo[mask][:,None], axis=0))
nrm = mat / sf[None,:]
cm = nrm[:, :len(ctrl)].mean(1); km = nrm[:, len(ctrl):].mean(1)
ct['baseMean'] = (cm + km) / 2
ct['my_lfc']   = np.log2((km + 1) / (cm + 1))
ct = ct[ct['baseMean'] > 5]
my_lfc  = dict(zip(ct['gene'].astype(str), ct['my_lfc']))
my_base = dict(zip(ct['gene'].astype(str), ct['baseMean']))
def _lfc(g):  return deg_lfc.get(g,  my_lfc.get(g))
def _base(g): return deg_base.get(g, my_base.get(g))
print(f'genes with baseMean>5: {len(ct)}')

3 · Expression-matched Random set via epi.utils.expression_matched_sample¶

Decile-bin matching on baseMean of 4C-activated genes; Random ends up ~same size as the activated set. Without expression matching the Random +/- p-value drops to 0.05, far from the paper's 0.8.

expressed = set(ct['gene'].astype(str))

# Drop any with missing baseMean (rare; DEG table occasionally has NaN)

pool     = [(g, b) for g, b in pool     if b is not None and np.isfinite(b)]

random_genes = set(epi.utils.expression_matched_sample(
    np.array([b for _, b in act_pool]),
    np.array([b for _, b in pool]),
    [g for g, _ in pool],
    n_bins=10, seed=0,
))
print(f'Random (expression-matched): {len(random_genes)}')

gtf = epi.utils.get_gene_annotation(str(GENCODE_GTF))
gtf['tss'] = np.where(gtf['strand'] == '+', gtf['start'], gtf['end'])
gene2loc = gtf.drop_duplicates('gene_name').set_index('gene_name')[['chrom','tss']]

s7 = pd.read_excel(SUPP_TABLES, sheet_name='Table S7_4C OTX2_CUTRUN_peaks',
                     header=None, names=['chrom','start','end','name','score'])
s7 = s7.dropna(subset=['chrom','start','end'])
peaks_by_chrom = {c: g[['start','end']].astype(int).values
                    for c, g in s7.groupby('chrom')}

def _bound(g):
    if g not in gene2loc.index: return None
    c, t = gene2loc.loc[g]
    arr = peaks_by_chrom.get(c)
    if arr is None: return False
    return bool(((arr[:,0] < t+2500) & (arr[:,1] > t-2500)).any())

def _split(genes):
    plus, minus = [], []
    for g in genes:
        v = _lfc(g); pb = _bound(g)
        if v is None or pb is None: continue
        (plus if pb else minus).append(v)
    return np.asarray(plus), np.asarray(minus)

p4, n4 = _split(activated_4c)
pr, nr = _split(random_genes)
print(f'4C: +{len(p4)} / -{len(n4)}   Random: +{len(pr)} / -{len(nr)}')

from scipy.stats import mannwhitneyu
from matplotlib.patches import Patch

P_4c   = mannwhitneyu(p4, n4, alternative='two-sided').pvalue
P_rand = mannwhitneyu(pr, nr, alternative='two-sided').pvalue
print(f'4C + vs -:   P = {P_4c:.2g}  (paper 2e-5)')
print(f'Rand + vs -: P = {P_rand:.2g}  (paper 0.8)')

fig, ax = plt.subplots(figsize=(3.6, 3.3))
data = [p4, n4, pr, nr]; positions = [1, 1.8, 3, 3.8]
cols = ['#C93838', '#3A6EB3'] * 2
parts = ax.violinplot(data, positions=positions, widths=0.72,
                        showmeans=False, showextrema=False)
for pc, col in zip(parts['bodies'], cols):
    pc.set_facecolor(col); pc.set_edgecolor('black'); pc.set_alpha(0.85); pc.set_linewidth(0.6)
